const DRAGON_EGG_STAGE_THRESHOLDS = [20, 50, 100];

const DRAGON_EGG_STAGE_NAMES = [
  "龙蛋（休眠）",
  "龙蛋（微光）",
  "龙蛋（裂纹）",
  "龙蛋（孵化）"

**2. 成长更新逻辑**
```javascript
function updateDragonEggGrowth(state, source, result) {
  const pointsToAdd = DRAGON_EGG_GROWTH_POINTS[source][result] || 0;
  if (pointsToAdd <= 0) return;

  state.dragonEgg.points += pointsToAdd;

  // 计算新阶段
  let newStage = 0;
  for (let i = 0; i < DRAGON_EGG_STAGE_THRESHOLDS.length; i++) {
    if (state.dragonEgg.points >= DRAGON_EGG_STAGE_THRESHOLDS[i]) {
      newStage = i + 1;
    }
  }
  state.dragonEgg.stage = newStage;

  // 显示反馈
  showFloatingText(`🐉 龙蛋成长 +${pointsToAdd}`, ...);
  if (newStage > oldStage) {
    showToast(`🥚 龙蛋升级：${DRAGON_EGG_STAGE_NAMES[newStage]}`);
  }
}
```

**3. 集成到事件记录**
```javascript

  // M3: 更新龙蛋成长

  saveProgress();
  return event;
}
```
